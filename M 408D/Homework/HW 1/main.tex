\begin{problem}
\begin{equation*}
\int^{1}_{0} x(1 - x^2)^4 dx  
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
u &= 1 - x^2 \\
\frac{du}{dx} &= -2x \\
du &= -2x dx
\end{align*}
\begin{align*}
\int^{1}_{0} x(1 - x^2)^4 dx &= -\frac{1}{2} \int^{1}_{0} -2x(1 - x^2)^4 dx \\
                             &= -\frac{1}{2} \int^{0}_{1} u^4 du \\
                             &= \frac{1}{2} \bigg| \frac{1}{5}u^5 \bigg|_0^1 \\
                             &= \boxed{\frac{1}{10}}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int^{\pi/2}_{0} \sin x(2f'(\cos x) - 1) dx
\end{equation*}
when $f(0) = 2$ and $f(1) = 4$
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
  u &= \cos x \\
  du &= -\sin x dx
\end{aligned} \\
\int^{\pi/2}_{0} \sin x(2f'(\cos x) - 1) dx &= -\int^{\pi/2}_{0} 2 f'(u) - 1 du \\
&= -2f(u)\bigg| + u\bigg| \\
&= -2f(\cos x)\bigg|_{0}^{\pi/2} +\cos x\bigg|_{0}^{\pi/2} \\
&= -2f(\cos(\pi/2)) + 2f(\cos(0)) + \cos(\pi/2) - \cos(0) \\
&= -2(2) + 2(4) + 0 - 1 \\
&= \boxed{3}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int^{\pi/4}_{0}\left(\frac{1}{\cos^2\theta} + 2\cos2\theta \right)d\theta
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\int^{\pi/4}_{0}\left(\frac{1}{\cos^2\theta} + 2\cos2\theta \right)d\theta &= \int^{\pi/4}_{0}(\sec^2\theta+ 2\cos2\theta)d\theta \\
&= \int^{\pi/4}_{0}\sec^2\theta d\theta + \int^{\pi/4}_{0}2\cos2\theta d\theta \\
&= \tan\theta\bigg|_{0}^{\pi/4} + \sin2\theta\bigg|_{0}^{\pi/4} \\
&= 1 - 0 + 1 - 0 = \boxed{2}
\end{align*}
\end{soln} 

\begin{problem}
\begin{equation*}
\int^{2}_{1}6\ln(3x)dx
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
  u &= 6\ln(3x) \\
  du &= \frac{6}{x}dx \\
  dv &= dx \\
  v &= x \\
\end{aligned} \\
\int^{2}_{1}6\ln(3x)dx &= \int^{2}_{1}udv = uv - \int vdu \\
                       &= 6x\ln(3x)\bigg|_1^2 - \int^{2}_{1}6dx \\
&= (6x\ln(3x) - 6x)\bigg|_1^2 \\
&= 12\ln6 - 12 - 6\ln3 + 6 \\
&= 12\ln6 - 6\ln3 - 6 \\
&= 6(2\ln6 - \ln3 - 1) \\
&= 6 \left(\ln \frac{36}{3} - 1 \right) \\
&= \boxed{6(\ln12 - 1)}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int \cos(\ln(x)) dx  
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
  u &= \cos(\ln(x)) \\
  du &= -\frac{1}{x}\sin(\ln(x)) dx \\
  dv &= dx \\
  v &= x \\
\end{aligned} \\
\int \cos(\ln(x)) dx &= uv - \int vdu \\
                     &= x\cos(\ln(x)) + \int \sin(\ln(x)) dx \\
\begin{aligned}
u &= \sin(\ln(x)) \\
du &= \frac{1}{x}\cos(\ln(x)) dx \\
dv &= dx \\
v &= x
\end{aligned} \\
                     &= x\cos(\ln(x)) + uv - \int vdu \\
                     &= x\cos(\ln(x)) + x\sin(\ln(x)) - \int \cos(\ln(x)) dx \\
2 \int \cos(\ln(x)) &= x\cos(\ln(x)) + x\sin(\ln(x)) \\
\int \cos(\ln(x)) &= \boxed{\frac{1}{2}x(\cos(\ln(x)) + \sin(\ln(x)))}
\end{align*}
\end{soln}

\begin{problem}
If $f$ is a continuous function such that
\begin{equation*}
\int^{16}_{0} f(x)dx = 16  
\end{equation*}
determine the value of the integral
\begin{equation*}
\int^{4}_{0} 6f(4x)dx
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
  u &= 4x \\
  du &= 4dx
\end{aligned} \\
\int^{4}_{0}6f(4x)dx &= \int^{16}_{0}6f(u)\frac{du}{4} \\
                     &= \frac{3}{2} \int^{16}_{0}f(u)du \\
                     &= \frac{3}{2}(16) = \boxed{24}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int^{\pi/4}_{0}6x\sec^2xdx  
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
  u &= 6x \\
  du &= 6dx \\
  dv &= \sec^2xdx \\
  v &= \tan x
\end{aligned} \\
\int^{\pi/4}_{0}6x\sec^2xdx &= uv - \int vdu \\
                            &= 6x\tan x\bigg|_{0}^{\pi/4} - \int^{\pi/4}_{0} 6\tan x dx \\
                            &= 6x\tan x\bigg|_{0}^{\pi/4} - 6\int^{\pi/4}_{0} \frac{\sin x}{\cos x} dx \\
\begin{aligned}
u &= \cos x \\
du &= -\sin x dx
\end{aligned} \\
                            &= 6x\tan x\bigg|_{0}^{\pi/4} + 6 \int^{\sqrt 2/2}_{1} \frac{1}{u} du \\
                            &= 6x\tan x\bigg|_{0}^{\pi/4} + 6\ln(u)\bigg|^{\sqrt 2/2}_{1} \\
                            &= \frac{3\pi}{2}(1) - 0 + 6\ln(\sqrt 2/2) - 0 \\
                            &= \frac{3\pi}{2} + 6\ln(\sqrt 2) - 6\ln(2) \\
                            &= \frac{3\pi}{2} + 3\ln(2) - 6\ln(2) \\
                            &= \boxed{\frac{3\pi}{2} -3\ln(2)}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int^{1}_{0} 4\sin(5\sqrt x)dx
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
u &= \sqrt x \\
u^2 &= x \\
2u du &= dx
\end{aligned} \\
\int^{1}_{0} 4\sin(5\sqrt x)dx &= \int^{1}_{0}8u\sin(5u)du \\
\begin{aligned}
  w &= 8u \\
  dw &= 8du \\
  dv &= \sin(5u) du \\
  v &= -\frac{1}{5}\cos(5u) 
\end{aligned} \\
                               &= -\frac{8}{5}u\cos(5u) + \int \frac{8}{5}\cos(5u)du \\
                               &= -\frac{8}{5}u\cos(5u)\bigg|_0^1 + \frac{8}{25}\sin(5u)\bigg|_0^1 \\
                               &= \boxed{-\frac{8}{5}\cos(5) + \frac{8}{25}\sin(5)}
\end{align*}
\end{soln}

\begin{problem}
Evaluate the integral 
\begin{equation*}
\int^{2}_{1} x^2 f''(x) dx
\end{equation*}
when $f(1) = 9$, $f(2) = 7$,
$f'(1) = 8$, $f'(2) = 6$
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
u &= x^2 \\
du &= 2x dx \\
dv &= f''(x) dx \\
v &= f'(x)
\end{aligned} \\
\int^{2}_{1} x^2 f''(x) dx &= x^2f'(x) - \int 2xf'(x) dx \\
\begin{aligned}
u &= 2x \\
du &= 2 dx \\
dv &= f'(x) dx \\
v &= f(x)
\end{aligned} \\
                           &= x^2f'(x) - 2xf(x) + \int 2f(x) dx \\
                           &= (x^2f'(x) - 2xf(x))\bigg|_{1}^{2} + \int^{2}_{1} 2f(x) dx \\
                           &= 4(6) - 4(7) - 1(8) - 2(9) + \int^{2}_{1} 2f(x) dx \\
                           &= \boxed{6 + 2\int^{2}_{1} f(x) dx}
\end{align*}
\end{soln}

\begin{problem}
\begin{equation*}
\int^{3}_{e} \frac{\ln x}{x^2} dx
\end{equation*}
\end{problem}

\begin{soln}
\begin{align*}
\begin{aligned}
u &= \ln x \\
du &= \frac{1}{x} dx \\
dv &= \frac{1}{x^2} dx \\
v &= -\frac{1}{x}
\end{aligned} \\
\int^{3}_{e} \frac{\ln x}{x^2} &= -\frac{\ln x}{x}\bigg| + \int \frac{1}{x^2}dx \\
                               &= \left( -\frac{\ln x}{x} - \frac{1}{x} \right)\bigg|_{e}^{3} \\
                               &= -\frac{\ln 3}{3} - \frac{1}{3} + \frac{1}{e} + \frac{1}{e} \\
                               &= \boxed{\frac{2}{e} - \frac{1}{3} (\ln 3 + 1)}
\end{align*}
\end{soln}
